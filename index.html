🛠️ Implementation Code for Your Website

📝 Note: This counter will start from 1145 and increment by 1 for each new visitor. The count is stored in the browser's localStorage.

HTML + CSS + JavaScript (Complete Solution):

<!-- Add this CSS to your existing stylesheet or in a <style> tag -->
<style>
.visitor-counter {
    background: linear-gradient(135deg, #4CAF50, #45a049);
    color: white;
    padding: 20px;
    border-radius: 12px;
    text-align: center;
    margin: 20px 0;
    box-shadow: 0 6px 20px rgba(76, 175, 80, 0.3);
    position: relative;
    overflow: hidden;
}

.visitor-counter::before {
    content: '';
    position: absolute;
    top: -50%;
    left: -50%;
    width: 200%;
    height: 200%;
    background: linear-gradient(45deg, transparent 30%, rgba(255,255,255,0.1) 50%, transparent 70%);
    animation: shine 4s infinite;
}

@keyframes shine {
    0% { transform: translateX(-100%) translateY(-100%) rotate(45deg); }
    100% { transform: translateX(100%) translateY(100%) rotate(45deg); }
}

.counter-content {
    position: relative;
    z-index: 2;
}

.counter-number {
    font-size: 2.5em;
    font-weight: bold;
    color: #fff;
    text-shadow: 2px 2px 4px rgba(0,0,0,0.3);
    font-family: 'Courier New', monospace;
}

.counter-label {
    font-size: 1.1em;
    margin-top: 8px;
    opacity: 0.9;
}
</style>

<!-- Add this HTML where you want the counter to appear -->
<div class="visitor-counter">
    <div class="counter-content">
        <div class="counter-number" id="visitor-count">1145</div>
        <div class="counter-label">👥 Total Visitors</div>
    </div>
</div>

<script>
// Visitor counter starting from 1145
function updateVisitorCount() {
    const INITIAL_COUNT = 1145; // Starting point
    const STORAGE_KEY = 'swachhta_visitor_count';
    
    // Get stored count or start from initial count
    let storedCount = localStorage.getItem(STORAGE_KEY);
    
    if (storedCount === null) {
        // First time visitor - initialize with base count
        localStorage.setItem(STORAGE_KEY, INITIAL_COUNT.toString());
        displayCount(INITIAL_COUNT);
    } else {
        // Returning visitor - increment count
        let currentCount = parseInt(storedCount) + 1;
        localStorage.setItem(STORAGE_KEY, currentCount.toString());
        displayCount(currentCount);
    }
}

function displayCount(count) {
    const counterElement = document.getElementById('visitor-count');
    if (counterElement) {
        // Animate the counter
        animateCounter(counterElement, count);
    }
}

function animateCounter(element, targetValue) {
    const duration = 2000;
    const startValue = 0;
    const increment = targetValue / (duration / 16);
    let currentValue = startValue;
    
    const timer = setInterval(() => {
        currentValue += increment;
        if (currentValue >= targetValue) {
            currentValue = targetValue;
            clearInterval(timer);
        }
        element.textContent = Math.floor(currentValue);
    }, 16);
}

// Initialize counter when page loads
document.addEventListener('DOMContentLoaded', updateVisitorCount);
</script>

Simple Version (Minimal Code):

<div style="background: #4CAF50; color: white; padding: 20px; border-radius: 10px; text-align: center; margin: 20px 0;">
    <h3>👥 Visitors: <span id="visitor-count">1145</span></h3>
</div>

<script>
function updateVisitorCount() {
    const INITIAL_COUNT = 1145;
    let count = localStorage.getItem('swachhta_visitors');
    
    if (count === null) {
        count = INITIAL_COUNT;
    } else {
        count = parseInt(count) + 1;
    }
    
    localStorage.setItem('swachhta_visitors', count);
    document.getElementById('visitor-count').textContent = count;
}

document.addEventListener('DOMContentLoaded', updateVisitorCount);
</script>
🚀 Quick Setup:
  1. Copy either code version above
  2. Paste it into your index.html file
  3. Change 1145 to your desired starting number
  4. Save and commit to GitHub
  5. Your counter will start from 1145 and increment for each new visitor!
⚙️ Customization Options:
  • Starting Number: Change INITIAL_COUNT = 1145 to any number
  • Storage Key: Change 'swachhta_visitors' to make it unique
  • Colors: Modify the background color from #4CAF50
  • Animation: Adjust animation duration or remove it entirely
